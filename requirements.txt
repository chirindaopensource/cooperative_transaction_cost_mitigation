# NumPy: High-performance multidimensional array operations and vectorized
# numerical computing. CRITICAL OPERATIONS:
#   - Eigenvalue decomposition for low-rank factor model (J=15 factors)
#   - Diagonal scaling matrix D construction from Hessian approximations
#   - Vectorized NAV-weighted aggregation of PM trade vectors: z = Σλ_i·x_i

# current solver bindings. DO NOT UPGRADE without verifying CVXPY compatibility.

# SciPy: Advanced scientific computing and linear algebra. CRITICAL OPERATIONS:

#     stationary covariance calibration in VAR(1) alpha noise generation
#   - scipy.linalg.eig: Eigendecomposition for factor loading extraction
#   - scipy.linalg.cholesky: Cholesky factorization for multivariate normal sampling
#   - BLAS/LAPACK bindings: Optimized matrix multiplication for daily covariance updates
# VERSION RATIONALE: 1.12.0 ensures compatibility with NumPy 1.26.x and provides
# stable Lyapunov solvers without numerical precision loss in high-dimensional settings.

#   - GroupBy operations: Cross-sectional daily rebalancing and IC calculations
#   - Rolling window statistics: 42-day forward volatility and covariance estimation
#   - State accounting: Track portfolio weights w_curr, cash balances c, and NAV λ_i

# (critical for daily cross-sectional analytics) and memory-efficient copy-on-write semantics.
pandas==2.2.1

# ------------------------------------------------------------------------------
# SECTION 2: DISCIPLINED CONVEX PROGRAMMING AND OPTIMIZATION
# ------------------------------------------------------------------------------
# CVXPY provides the domain-specific language (DSL) for formulating and solving
# the PM-level and Planner-level convex optimization subproblems in the ADMM
# decomposition. It abstracts solver backends and ensures DCP compliance.

# CVXPY: Disciplined Convex Programming framework. CRITICAL CAPABILITIES:
#   - DCP verification: Ensures objective and constraints satisfy convexity rules
#   - Proximal operator representation: Indicator functions I_X and I_Z for constraints
#   - Problem canonicalization: Transforms high-level specification into solver-compatible form
#   - Warm-starting: Reuses dual variables across ADMM iterations for faster convergence
#   - Diagonal scaling: Supports D-matrix preconditioning for better conditioning
# SUBPROBLEMS IMPLEMENTED:
#   - PM-level: min_x (1/2)·x^T·Σ·x - α^T·x + (ρ/2)·||x - z + u/ρ||^2 + I_X(x)
#   - Planner-level: min_z κ(z) + (ρM/2)·||z - x_avg - u_avg/ρ||^2 + I_Z(z)
# VERSION RATIONALE: 1.5.1 provides robust DPP (disciplined parametrized programming)
# support for time-varying parameters (α_t, Σ_t, κ_t) across 6,300 daily rebalances.
cvxpy==1.5.1

# ------------------------------------------------------------------------------
# SECTION 3: CONVEX OPTIMIZATION SOLVER BACKENDS
# ------------------------------------------------------------------------------
# CVXPY requires at least one installed solver. For institutional-grade performance
# with thousands of constraints and decision variables, we specify a diversified
# solver ecosystem with complementary strengths.

# ECOS: Embedded Conic Solver for second-order cone programs (SOCP).
# USAGE: Small-to-medium PM-level subproblems with O(100-500) decision variables
# and O(500-2000) constraints (leverage, concentration, turnover limits).
# STRENGTHS: Fast warm-starting, low memory overhead, stable for well-conditioned problems.
# VERSION RATIONALE: 2.0.13 provides deterministic behavior for reproducible benchmarks.
ecos==2.0.13

# SCS: Splitting Conic Solver via Douglas-Rachford operator splitting.
# USAGE: Large-scale Planner-level problems aggregating M PM trade vectors.
# STRENGTHS: Handles ill-conditioned problems, robust dual infeasibility detection,
# optional GPU acceleration via scs[gpu] extra (requires CUDA toolkit).
# VERSION RATIONALE: 3.2.4.post1 includes adaptive step-size scaling and improved
# termination criteria for non-stationary market environments.
scs==3.2.4.post1

# OSQP: Operator Splitting Quadratic Program solver.
# USAGE: Quadratic subproblems arising from 3/2-power cost approximation.
# STRENGTHS: Highly efficient warm-starting, adaptive rho tuning (complements manual ρ=10.0),
# polishing for high-accuracy solutions required for transaction cost accounting.
# VERSION RATIONALE: 0.6.7 provides numerically stable polishing for 1e-8 tolerance.
osqp==0.6.7

# CLARABEL: Modern conic interior-point solver with hybrid step direction search.
# USAGE: Fallback for challenging instances with near-degenerate constraints.
# STRENGTHS: Excellent numerical stability, infeasibility certificates for debugging,
# presolve analysis for automatic problem reduction, handles both primal and dual infeasibility.
# VERSION RATIONALE: 0.7.1 is the latest stable release with proven reliability in production.
clarabel==0.7.1
